repo-url:
  description: "Repo URL for GitHub scan"
  required: false
  type: string
  default: ""



- name: Determine repo URL
  id: repo
  run: |
    if [ -n "${{ inputs.repo-url }}" ]; then
      echo "url=${{ inputs.repo-url }}" >> $GITHUB_OUTPUT
    else
      echo "url=https://github.com/${{ github.repository }}" >> $GITHUB_OUTPUT
    fi



# -------------------------
# 4. Scan GitHub Issues & PR Comments
# -------------------------
- name: Scan GitHub Issues & PR Comments
  run: |
    echo "Scanning GitHub issues and PR comments..."

    docker run --rm \
      -e GITHUB_TOKEN=${{ secrets.TRUFFLEHOG_TOKEN }} \
      trufflesecurity/trufflehog:latest \
      github \
        --repo=${{ steps.repo.outputs.url }} \
        --issue-comments \
        --pr-comments \
        --token=${{ secrets.TRUFFLEHOG_TOKEN }} \
        --json > trufflehog-github-results.json || true

    echo "GitHub scan completed"



# -------------------------
# 5. Merge Results
# -------------------------
- name: Merge Scan Results
  run: |
    echo "Merging scan results..."

    if [ -f trufflehog-github-results.json ]; then
      cat trufflehog-results.json trufflehog-github-results.json > combined-results.json
      mv combined-results.json trufflehog-results.json
    fi



SECRET_COUNT=$(jq length trufflehog-results.json 2>/dev/null || echo "0")
